#include <iostream>
#include <string>
#include<cmath>
using namespace std;

int main()
{
    double pi4 = 0.;
    long n;
    double dif;
    double a=1;
    double b;
    cout << "Number of iterations? ";
    cin >> n;
    n=n/2;
  
  for ( n > 1; n --;) {
      b=a+2;
      dif=(1/a)-(1/b);
      a=a+4;
      pi4+=dif;
    }

    
    cout.precision(20);
    cout << "Pi = " << (pi4 * 4.) << endl;
    return 0;
}
