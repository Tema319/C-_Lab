#include <iostream>
#include <string>
#include<cmath>
using namespace std;

int main() {
    double x, y;

    cout << "Lat's count y = sin^5(x) + |5x - 1.5|" << endl;
    cout << "Enter 5 values of x:\n";

    for (int i = 1; i <= 5; i++) {
        cout << "x" << i << " = ";
        cin >> x;

        y = pow(sin(x), 5) + fabs(5 * x - 1.5);

        cout << "y" << i << " = " << y << endl;
    }

}
