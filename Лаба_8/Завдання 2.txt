#include <iostream>
#include <string>
#include<cmath>
using namespace std;

int main()
{
  int m;  
 
 do{
  cout<<"Enter how big is array:";
  cin>>m;
    } while (m==0); 

int a[m]; 

  for(int i = 0; i < m; i++){
      cout<<"Enter value:";
      cin>>a[i];
      }

 int max = a[0];
 
 for (int i = 0; i < m; i ++) {
        if (max<a[i]){
            max=a[i];
            }
}

 int x=0;
 
while (a[x]!=max){
  if (a[x]<0){
      a[x]=0;
      }
     x++;
    }  

cout<<"Array is:";

 for (int i = 0; i < m; i++) {
        cout << a[i] << " ";
 }
 
}
